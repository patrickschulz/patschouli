\pgfplotsset{
    compat/show suggested version = false,
    compat=1.15,
    % settings
    every axis/.append style={
        grid=both, 
        enlarge x limits = false, 
        enlarge y limits = true, 
        %rotate ylabel, 
        width=0.7\linewidth, 
        height=5cm,
        scale only axis,
        scaled ticks=false,
        axis line style = {thick},
        tick align = inside,
        major tick length = {0.12cm},
        minor tick length = {0.08cm},
        every tick/.append style = {semithick, black},
        ylabel style = {minimum height = 1cm}
    },
    every axis plot/.style={thick},
    label style={font=\normalsize},
    tick label style={font=\normalsize},
    grid style={black, line width = 0.3pt, dash pattern = on 0pt off 1.2pt, line cap = round},
    legend cell align=left,
    legend pos={south west},
    legend style = {draw = none},
    %cycle list name = linestyles,
    cycle list = {solid, densely dashed, densely dotted},
    % styles to be used
    textbook/.style = {
        axis x line = middle, 
        axis y line = middle, 
        xlabel style = {at={(xticklabel* cs:1)}, right}, 
        ylabel style = {at={(yticklabel* cs:1)}, above},
    },
    datagraph/.style = {
        axis lines = box
    },
    overlay/.style = {
        xtick=\empty,
        axis y line* = right,
        grid = none
    },
    rotate ylabel/.style={ylabel style={rotate=-90}},
    markline/.style={thick, densely dotted, black},
    specialline/.style={thick, dashed, black},
    fillplot/.style={fill=#1, draw=none, fill opacity=0.2},
    dirac/.style={
        mark=triangle*,
        mark options={scale=0.75},
        ycomb,
        scatter,
        visualization depends on={y/abs(y)-1 \as \sign},
        scatter/@pre marker code/.code={\scope[rotate=90*\sign,yshift=-2pt]}
    },
    spectrum/.style={
        ycomb,
        mark = none
    },
    % clear ticks
    clear ticklabels/.style = {xticklabels = {}, yticklabels = {}},
    clear xticklabels/.style = {xticklabels = {}},
    clear yticklabels/.style = {yticklabels = {}},
    todo/.style = {
        axis line style = {red}
    },
    % preliminary picture
    preliminary/.style = {
        axis line style = {red}
    }
}
